-- === Vòng lặp kiểm tra âm nhạc & Text HUD ===
task.spawn(function()
    local lastSound = nil
    local endedConnection = nil

    while task.wait(0.5) do
        if CurrentMusic then
            -- Cập nhật Text
            local sName = CurrentSelectedSongName
            if sName == "" or sName == nil then sName = "Không xác định" end
            MusicHUDLabel.Text = sName -- Đã bỏ Emoji cũ ở đây

            -- Khi bài hát chuyển sang mới
            if CurrentMusic ~= lastSound then
                lastSound = CurrentMusic
                CurrentMusic.Looped = IsLooped -- Cập nhật Looped theo chế độ
                
                -- Ngắt kết nối Ended cũ để tránh đụng logic
                if endedConnection then
                    endedConnection:Disconnect()
                    endedConnection = nil
                end
                
                -- Gắn sự kiện Ended mới để phát ngẫu nhiên NẾU đang tắt Lặp (Loop = false)
                endedConnection = CurrentMusic.Ended:Connect(function()
                    if not IsLooped then
                        task.spawn(function()
                            local keys = {}
                            for k, v in pairs(MusicDatabase) do table.insert(keys, k) end
                            local randomKey = keys[math.random(1, #keys)]
                            
                            -- Tránh phát lại đúng bài cũ nếu playlist lớn hơn 1
                            while randomKey == CurrentSelectedSongName and #keys > 1 do
                                randomKey = keys[math.random(1, #keys)]
                            end
                            
                            CurrentSelectedSongName = randomKey
                            PlayMusic(MusicDatabase[randomKey], randomKey)
                        end)
                    end
                end)
            else
                -- Đảm bảo trạng thái Looped luôn đúng nếu user bấm Toggle giữa chừng
                if CurrentMusic.Looped ~= IsLooped then
                    CurrentMusic.Looped = IsLooped
                end
                
                -- FALLBACK BẢO VỆ
                if not IsLooped and not CurrentMusic.IsPlaying and CurrentMusic.TimeLength > 0 and (CurrentMusic.TimeLength - CurrentMusic.TimePosition) < 1.5 then
                    task.spawn(function()
                        local keys = {}
                        for k, v in pairs(MusicDatabase) do table.insert(keys, k) end
                        local randomKey = keys[math.random(1, #keys)]
                        while randomKey == CurrentSelectedSongName and #keys > 1 do
                            randomKey = keys[math.random(1, #keys)]
                        end
                        CurrentSelectedSongName = randomKey
                        PlayMusic(MusicDatabase[randomKey], randomKey)
                    end)
                end
            end
        else
            MusicHUDLabel.Text = "Không Có" -- Sửa theo yêu cầu
            lastSound = nil
            if endedConnection then
                endedConnection:Disconnect()
                endedConnection = nil
            end
        end
    end
end)

        _G._scriptVersion = _G._scriptVersion or "BETA"

        _G._lastFeedbackSend = _G._lastFeedbackSend or 0
        local now = tick()
        local RATE_LIMIT = 30 -- giây

        if now - _G._lastFeedbackSend < RATE_LIMIT then
            local waitSec = math.ceil(RATE_LIMIT - (now - _G._lastFeedbackSend))
            if Notify then
                pcall(Notify, "Vui lòng đợi", ("Bạn vừa gửi rồi — thử lại sau %d giây."):format(waitSec))

                end)

            if Notify then
                pcall(Notify, "Lỗi", "Vui lòng nhập nội dung trước khi gửi.")

                end)
            end
            return
        end

        -- Parse tags (hỗ trợ [tag] hoặc #tag)
        local function extract_tag(s, candidates)
            s = s:lower()
            for _, c in ipairs(candidates) do
                if s:find("%["..c.."%]") or s:find("#"..c) or s:find("%("..c.."%)") then
                    return c
                end
            end
            return nil
        end

        local function remove_all_tags(s)
            -- loại bỏ các dạng [tag], #tag, (tag)
            s = s:gsub("%[(.-)%]", ""):gsub("#(%w+)", ""):gsub("%((.-)%)", "")
            -- trim
            s = s:gsub("^%s+", ""):gsub("%s+$", "")
            return s
        end

        local typeCandidates = {"bug", "idea", "other"}
        local prioCandidates = {"low", "normal", "high"}

        local lowerText = text:lower()
        local detectedType = extract_tag(lowerText, typeCandidates) or "other"
        local detectedPrio = extract_tag(lowerText, prioCandidates) or "normal"

        -- chuẩn hoá display
        local displayType = ({bug="Bug", idea="Idea", other="Other"})[detectedType] or "Other"
        local displayPrio = ({low="Low", normal="Normal", high="High"})[detectedPrio] or "Normal"

        local cleaned = remove_all_tags(text)

        local MAX_DESC = 1900
        if #cleaned > MAX_DESC then

                    -- Tắt Post-Processing, Mây, Khí quyển

                        if child:IsA("PostEffect") or child:IsA("BlurEffect") or child:IsA("SunRaysEffect") or child:IsA("ColorCorrectionEffect") or child:IsA("BloomEffect") or child:IsA("DepthOfFieldEffect") or child:IsA("Atmosphere") or child:IsA("Clouds") then
                            -- Lưu trạng thái để khôi phục
                            if child:IsA("Atmosphere") or child:IsA("Clouds") then
                                _originalState.lighting.effects[child] = { Parent = child.Parent }
                                pcall(function() child.Parent = nil end) -- Cất đi thay vì xóa
                            else
                                _originalState.lighting.effects[child] = { Enabled = child.Enabled }
                                pcall(function() child.Enabled = false end)
                            end
                        end
                    end

                    lighting.GlobalShadows = false    
                    lighting.FogEnd = 9e9
                    lighting.EnvironmentDiffuseScale = 0
                    lighting.EnvironmentSpecularScale = 0
                end)

                -- [2] TỐI ƯU TERRAIN (CỎ & NƯỚC - Boost FPS rất mạnh)
                if terrain then
                    pcall(function()
                        _originalState.terrain = {
                            Decoration = terrain.Decoration,
                            WaterWaveSize = terrain.WaterWaveSize,
                            WaterWaveSpeed = terrain.WaterWaveSpeed
                        }
                        terrain.Decoration = false -- TẮT CỎ 3D
                        terrain.WaterWaveSize = 0    
                        terrain.WaterWaveSpeed = 0 
                    end)
                end

                -- [3] TỐI ƯU PART & HIỆU ỨNG (Chia nhỏ tiến trình chống treo máy)
                local count = 0
                for _, v in ipairs(workspaceDesc) do
                    -- BasePart
                    if v:IsA("BasePart") and not v:IsA("MeshPart") then
                        if not _originalState.parts[v] then
                            _originalState.parts[v] = {
                                Material = v.Material,
                                Reflectance = v.Reflectance,
                                Transparency = v.Transparency,
                                CastShadow = v.CastShadow
                            }
                        end
                        v.Material = Enum.Material.SmoothPlastic
                        v.Reflectance = 0
                        pcall(function() v.CastShadow = false end)

                        -- Mờ kính
                        if v.Name:lower():find("glass") or v.Name:lower():find("window") then
                            v.Transparency = math.max(v.Transparency, 0.25)
                        end

                    -- Decal / Texture
                    elseif v:IsA("Decal") or v:IsA("Texture") then
                        if not _originalState.decals[v] then
                            _originalState.decals[v] = { Transparency = v.Transparency }
                        end
                        v.Transparency = 1 -- Ẩn đi thay vì Destroy để có thể Restore

                    elseif v:IsA("ParticleEmitter") or v:IsA("Beam") or v:IsA("Trail") or v:IsA("Sparkles") or v:IsA("Fire") or v:IsA("Smoke") then
                        if not _originalState.parts[v] then
                            _originalState.parts[v] = { Enabled = v.Enabled }
                        end
                        v.Enabled = false

                    -- Âm thanh rác
                    elseif v:IsA("Sound") then
                        if not _originalState.parts[v] then
                            _originalState.parts[v] = { Volume = v.Volume, Playing = v.IsPlaying }
                        end
                        v.Volume = 0
                        if v.IsPlaying then pcall(function() v:Pause() end) end
                    end

                    -- BÍ QUYẾT CHỐNG TREO: Nghỉ 1 frame sau mỗi 500 object
                    count = count + 1
                    if count % 500 == 0 then
                        task.wait()
                    end

-- Status HUD (phiên bản đã tinh chỉnh)
local HUD_NAME = "67_Status_HUD_v1"
_G.ShowStatusHUD = _G.ShowStatusHUD or false

local statusGui, statusFrame, statusLabel
local connRender -- để lưu connection RenderStepped
local RunService = game:GetService("RunService")
local Players = game:GetService("Players")
local LocalPlayer = Players.LocalPlayer

-- Hỗ trợ tìm GUI parent an toàn (fallback nếu GetSafeGui() không tồn tại)
local function getSafeGui()
    if type(GetSafeGui) == "function" then
        local ok, g = pcall(GetSafeGui)
        if ok and g then return g end
    end
    -- Nếu LocalPlayer chưa sẵn sàng thì chờ; nếu chạy trong môi trường server thì dùng CoreGui (cẩn thận)
    if Players.LocalPlayer then
        local playerGui = Players.LocalPlayer:FindFirstChild("PlayerGui")
        if playerGui then return playerGui end
        -- fallback: wait for PlayerGui (local script)
        local ok, pg = pcall(function() return Players.LocalPlayer:WaitForChild("PlayerGui", 5) end)
        if ok and pg then return pg end
    end
    -- cuối cùng fallback vào CoreGui (có thể bị hạn chế trong một số môi trường)
    return game:GetService("CoreGui")
end

local function destroyStatusHUD()
    pcall(function()
        -- disconnect RenderStepped nếu có
        if connRender and connRender.Connected then
            connRender:Disconnect()
            connRender = nil
        end

        if statusGui and statusGui.Parent then
            statusGui:Destroy()
        end

        -- cố gắng xoá ở các nơi khác (nếu GetSafeGui tồn tại)    
        local safe = (type(GetSafeGui) == "function" and pcall(GetSafeGui) and getSafeGui()) or getSafeGui()    
        if safe and safe.FindFirstChild then    
            local existing = safe:FindFirstChild(HUD_NAME)  
            if existing then existing:Destroy() end    
        end    

        local coreExisting = game:GetService("CoreGui"):FindFirstChild(HUD_NAME)    
        if coreExisting then coreExisting:Destroy() end    

        statusGui, statusFrame, statusLabel = nil, nil, nil
    end)
end

_G.destroyStatusHUD = destroyStatusHUD

-- Biến lưu các nhãn (labels) tách biệt
local lblTime, lblFPS, lblPing

local function createStatusHUD()
    pcall(function()
        destroyStatusHUD()

        local safeGui = getSafeGui()

        statusGui = Instance.new("ScreenGui")    
        statusGui.Name = HUD_NAME    
        statusGui.IgnoreGuiInset = true    
        statusGui.ResetOnSpawn = false    
        statusGui.Parent = safeGui 

        statusFrame = Instance.new("Frame")    
        statusFrame.Size = UDim2.new(0, 320, 0, 24)    
        statusFrame.AnchorPoint = Vector2.new(1, 0)    
        statusFrame.Position = UDim2.new(1, -8, 0, 8)    
        statusFrame.BackgroundColor3 = Color3.fromRGB(0, 0, 0)    
        statusFrame.BackgroundTransparency = 0.35    
        statusFrame.BorderSizePixel = 0    
        statusFrame.ZIndex = 9999    
        statusFrame.Parent = statusGui    

        local corner = Instance.new("UICorner")    
        corner.CornerRadius = UDim.new(0, 6)    
        corner.Parent = statusFrame    

        -- [FIX] Sử dụng UIListLayout để cố định vị trí các nhãn
        local layout = Instance.new("UIListLayout", statusFrame)
        layout.FillDirection = Enum.FillDirection.Horizontal
        layout.HorizontalAlignment = Enum.HorizontalAlignment.Center
        layout.VerticalAlignment = Enum.VerticalAlignment.Center
        layout.SortOrder = Enum.SortOrder.LayoutOrder
        layout.Padding = UDim.new(0, 5)

        local function createFixedLabel(name, width, order)
            local lbl = Instance.new("TextLabel")
            lbl.Name = name
            lbl.Size = UDim2.new(0, width, 1, 0)
            lbl.BackgroundTransparency = 1
            lbl.Font = Enum.Font.GothamSemibold
            lbl.TextColor3 = Color3.fromRGB(255, 255, 255)
            lbl.TextSize = 13
            lbl.TextXAlignment = Enum.TextXAlignment.Center
            lbl.LayoutOrder = order
            lbl.ZIndex = 10000
            lbl.Parent = statusFrame
            return lbl
        end

        lblTime = createFixedLabel("Time", 85, 1)
        local sep1 = createFixedLabel("Sep1", 10, 2)
        sep1.Text = "|"
        lblFPS = createFixedLabel("FPS", 65, 3)
        local sep2 = createFixedLabel("Sep2", 10, 4)
        sep2.Text = "|"
        lblPing = createFixedLabel("Ping", 130, 5)

        local smoothed = 60    
        local startTime = tick()    
        local pingUpdateTimer = 2 -- Đặt là 2 để lấy dữ liệu ngay lần đầu tiên
        local lastFpsText = "60"
        local lastPingText = "??ms"
        local lastCv = 0

        if connRender and connRender.Connected then    
            connRender:Disconnect()    
            connRender = nil    
        end    

        connRender = RunService.RenderStepped:Connect(function(dt)    
            if not _G.ShowStatusHUD or not statusFrame or not statusFrame.Parent then    
                destroyStatusHUD()    
                return    
            end    

            -- Tính FPS ngầm để giữ độ chính xác
            if dt > 0 then    
                local fps = 1 / dt    
                smoothed = smoothed * 0.85 + fps * 0.15    
            end   

            -- Đo FPS và Ping mỗi 2 giây
            pingUpdateTimer = pingUpdateTimer + dt
            if pingUpdateTimer >= 2 then
                pingUpdateTimer = 0
                lastFpsText = tostring(math.floor(smoothed + 0.5))    

                -- Tính Ping
                local pingVal = nil    
                pcall(function()    
                    local stats = game:GetService("Stats")    
                    if stats and stats:FindFirstChild("Network") and stats.Network:FindFirstChild("ServerStatsItem") then     
                        local s = stats.Network.ServerStatsItem    
                        if s["DataPing"] then pingVal = math.floor(s["DataPing"]:GetValue() + 0.5) end    
                    end    
                end)    
                pcall(function()    
                    if (not pingVal or pingVal <= 0) and LocalPlayer and LocalPlayer.GetNetworkPing then  
                        local p = LocalPlayer:GetNetworkPing()    
                        if type(p) == "number" then pingVal = math.floor(p * 1000 + 0.5) end    
                    end    
                end)   
                lastPingText = pingVal and (tostring(pingVal) .. "ms") or "??ms"    

                -- Tính CV%
                local diff = 60 - smoothed    
                if diff > 0 then lastCv = math.floor(math.min(100, (diff / 60) * 100) + 0.5) else lastCv = 0 end    
            end

            -- Tính Time (cập nhật liên tục để thời gian trôi mượt mà)
            local now = tick()    
            local elapsed = math.floor(now - startTime + 0.5)    
            local hours = math.floor(elapsed / 3600)    
            local mins = math.floor((elapsed % 3600) / 60)    
            local secs = elapsed % 60    
            local timestr = string.format("%02d:%02d:%02d", hours, mins, secs)    

            -- Cập nhật Text tách biệt
            pcall(function()
                lblTime.Text = "Time: " .. timestr
                lblFPS.Text = "FPS: " .. lastFpsText
                lblPing.Text = string.format("Ping: %s (%02d%%CV)", lastPingText, lastCv)
            end)
        end)
    end)
end

-- helper to get LocalPlayer safely (waits up to 5s)
local function getLocal()

-- Safe clipboard setter (works across many executors)
local function safeSetClipboard(text)
    local ok, _ = pcall(function()
        if setclipboard then setclipboard(text) end
        if syn and syn.set_clipboard then syn.set_clipboard(text) end
        if set_clipboard then set_clipboard(text) end
        if clipboard and clipboard.set then clipboard.set(text) end
    end)
    return ok
end

-- Safe clipboard reader (tries common executor functions)
local function tryReadClipboard()
    local ok, result = pcall(function()
        if is_sirhurt_closure then -- example for some exploit environments
            -- no read support
            return nil
        end
        if getclipboard then return getclipboard() end
        if syn and syn.get_clipboard then return syn.get_clipboard() end
        if clipboard and clipboard.get then return clipboard.get() end
        -- no generic safe method; some executors expose nothing
        return nil
    end)
    if ok then return result end
    return nil
end

-- small helper to do HTTP requests in different executors (returns body string or nil)
local function doRequest(url)
    local req = (syn and syn.request) or (http and http.request) or http_request or request
    if req then
        local ok, res = pcall(function()
            return req({ Url = url, Method = "GET" })
        end)
        if ok and res then
            -- support different response field names
            if res.Body and res.Body ~= "" then return res.Body end
            if res.body and res.body ~= "" then return res.body end
        end
    end

    -- fallback to Roblox's HttpGet (may be disabled depending on environment)
    local ok2, body = pcall(function() return game:HttpGet(url, true) end)
    if ok2 and body and body ~= "" then return body end

    return nil
end

-- get a single page of public servers for this place
local function getServerPage(cursor)
    local url = "https://games.roblox.com/v1/games/" .. placeId .. "/servers/Public?limit=100"
    if cursor and cursor ~= "" then
        url = url .. "&cursor=" .. HttpService:UrlEncode(tostring(cursor))
    end
    local body = doRequest(url)
    if not body then return nil end
    local ok, data = pcall(function() return HttpService:JSONDecode(body) end)
    if ok and type(data) == "table" then return data end
    return nil
end

-- collect candidate servers (skips current job)
local function collectCandidateServers(maxPages)
    maxPages = tonumber(maxPages) or 8
    local servers = {}
    local cursor = nil
    for i = 1, maxPages do -- limit pages to avoid endless loops
        local page = getServerPage(cursor)
        if not page then break end
        if page.data and type(page.data) == "table" then
            for _, s in ipairs(page.data) do
                if s.id and tostring(s.id) ~= tostring(game.JobId) and s.playing and s.maxPlayers then
                    table.insert(servers, s)
                end
            end
        end
        cursor = page.nextPageCursor
        if not cursor or cursor == "" then break end
    end
    return servers
end

-- CHẠY LOADING TRƯỚC, RAYFIELD SAU (TỐI ƯU HÓA RE-EXECUTE)
----------------------------------------------------------------
local function RunLoading()
    -- NẾU LÀ RE-EXECUTE -> BỎ QUA LOADING, MỞ THẲNG MENU
    if isReExecution then
        if LoadingGui then LoadingGui:Destroy() end
        StartRayfield()
        -- Không gọi ShowGlobalAnnouncement() để chống spam
        return
    end

    -- NẾU LÀ CHẠY LẦN ĐẦU -> HIỂN THỊ LOADING GALAXY
    local steps = {
        {0.15, "Đang kiểm tra giao thức bảo mật..."},
        {0.35, "Đang tải tài sản hệ thống..."},
        {0.55, "Chèn các tập lệnh chính..."},
        {0.75, "Đồng bộ hóa các phần tử giao diện người dùng..."},
        {0.90, "Đang hoàn tất cấu hình..."},
        {1.0, "✨ Hệ thống đã sẵn sàng!"}
    }
    for _, s in ipairs(steps) do
        TweenService:Create(BarFill, TweenInfo.new(1.0, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {Size = UDim2.new(s[1], 0, 1, 0)}):Play()
        StatusText.Text = s[2]
        task.wait(0.6) 
    end
    
    task.wait(0.5)
  
    -- [FIX]: Nếu user vừa Execute thêm lần nữa trong lúc đang chờ Loading, tự động hủy luồng cũ này!
    if currentExecute ~= _G.Hizuma67_ExecuteCount then 
        return 
    end

    if LoadingGui then LoadingGui:Destroy() end
    StartRayfield()
    -- Chỉ gọi thông báo Global khi lần đầu khởi chạy
    pcall(function()
        ShowGlobalAnnouncement()
    end)
end

-- KÍCH HOẠT HỆ THỐNG
RunLoading()

-- [BẢN VÁ LỖI KẸT NÚT TSB TRÊN ĐIỆN THOẠI]
task.spawn(function()
    local Players = game:GetService("Players")
    local UserInputService = game:GetService("UserInputService")
    local CoreGui = game:GetService("CoreGui")

    local function FixMobileTouchBlocks(char)
        if char then pcall(function() char:WaitForChild("Humanoid", 5) end) end

        for i = 1, 3 do
            pcall(function()
                local player = Players.LocalPlayer
                if not player then return end

                if UserInputService.ModalEnabled then
                    UserInputService.ModalEnabled = false
                end

                local playerGui = player:FindFirstChild("PlayerGui")
                if playerGui then
                    local touchGui = playerGui:FindFirstChild("TouchGui")
                    if touchGui and touchGui:FindFirstChild("TouchControlFrame") then
                        touchGui.TouchControlFrame.Visible = true
                    end
                end

                for _, gui in ipairs(CoreGui:GetChildren()) do
                    if gui:IsA("ScreenGui") and tostring(gui.Name) == "LoadingGui" then
                        gui:Destroy()
                    end
                end
            end)
            task.wait(1) 
        end
    end

    local player = Players.LocalPlayer
    if player then
        if _G._tsbMobileFixConn then
            pcall(function() _G._tsbMobileFixConn:Disconnect() end)
        end
        _G._tsbMobileFixConn = player.CharacterAdded:Connect(function(char)
            task.spawn(FixMobileTouchBlocks, char)
        end)
        if player.Character then
            task.spawn(FixMobileTouchBlocks, player.Character)
        end
    end
end)
